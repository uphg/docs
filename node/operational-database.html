# 安装 Docker

  1. 注册：https://hub.docker.com/ (opens new window)
  2. 下载：Docker for Windows Installer (opens new window)
  3. 确认输入 docker --version 后返回版本号
  4. 设置国内镜像，教程 (opens new window)，镜像的地址为：https://docker.mirrors.ustc.edu.cn/
  5. 确保命令行输入 docker run hello-world 时输出 Hello from Docker!
  6. 如果无法安装新版，请查看旧版教程

新版配置镜像示例

# 旧版 Docker

  1. 下载 Docker Toolbox (opens new window)
  2. 打开 Docker QuickStart，运行后看到一个IP，退出（运行 docker-machine ip 可得到 IP）
  3. 重新打开命令行，输入 docker -v 查看是否正常
  4. 设置国内镜像，教程 (opens new window)，镜像的地址为：https://docker.mirrors.ustc.edu.cn/
  5. 确保命令行输入 docker run hello-world 时输出 Hello from Docker!

# 安装 mysql

创建一个 mysql 数据库

docker run --name mysql123 -e MYSQL_ROOT_PASSWORD=123456 -p 3306:3306 -d mysql:5.7.33

关于参数

--name   数据库名称
 -e      设置密码
 -p      设置端口
 -d      安装的 mysql 版本

# Docker 常用命令

docker ps                         # 查看运行状态
docker kill mysql123              # 关掉指定容器（指定名称或ID）
docker container start mysql123   # 开启刚刚关闭的容器
docker rm mysql123                # 删除指定容器，-f 强制

# 连接 mysql

  1. 进入容器
docker exec -it mysql123 bash
  1. 输入账户密码
mysql -u root -p
  1. 查看数据库列表
show databases; # 必须加分号

# 使用 Node.js 创建数据库

var mysql      = require('mysql');
var connection = mysql.createConnection({
  host     : 'localhost',
  user     : 'root',
  password : '123456'
});

connection.connect(); // 连接数据库

/* 创建一个名为 cheng 的数据库 */
connection.query('CREATE DATABASE IF NOT EXISTS cheng DEFAULT CHARSET utf8mb4 COLLATE utf8mb4_unicode_ci;', function (error, results, fields) {
  if (error) throw error;
  console.log('创建数据库');
  console.log(results);
});

/* 使用 cheng 数据库 */
connection.query('use cheng')
/* 创建一个名为 user 的表 */
connection.query(`CREATE TABLE IF NOT EXISTS user(
  name text,
  age int
)`, function (error, results, fields) {
  if (error) throw error;
  console.log('创建表');
  console.log(results);
});

connection.end();

# 操作表内容

# 在 user 表添加数据
insert into user (name, age) values ('ject', 18);

# 修改 user 表中 name 为 ject 的数据，将它的 age 字段改为 70
update user set age=70 where name='ject';

# 不要使用以下方式修改，会将数据库中所有 age字段都修改
update user set age=18;

# 查询 user 表所有字段
select * from user;

# 查询 user 表中的指定字段
select name,age from user;

# 只查询该字段的前十行
select name,age from user limit 10;

# 数据类型

  • 数字类型：https://dev.mysql.com/doc/refman/8.0/en/numeric-type-syntax.html (opens new window)
  • 字符串类型
  • 时间和日期类型，推荐关注 ISO 8601 及 如何把日期输出为 ISO 8601 格式 (opens new window)
  • JSON 类型（5.7.8以上版本）
  • 其他特殊类型

使用 serial 类型（数字类型）添加用户 id 案例

# 添加一列 id
alter table user add id serial;

# 查看列表
describe user;
