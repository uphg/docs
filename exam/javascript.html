# 2. 必考 Promise、Promise.all、Promise.race 分别怎么用？

Promise 使用

创建一个函数fn，它有一个参数，不传默认值为 1，它 return 一个 new Promise(function(resolve, reject){}) 函数，函数中有一个data，每次请求时 return 的 Promise 就会做一个判断，如果value的值为1，就让data的数据返回一个请求成功，如果value的值为2，就让data的数据返回一个请求失败。

function fn(value = 1){
    return new Promise(function(resolve, reject){
        let data
        if(value === 1){
            data = '请求成功'
            resolve(data)
        }else {
            data = '请求失败'
            reject(data)
        }
        
    })
}
// 成功请求案例
fn().then((data)=>{ console.log(data) },(error)=>{ console.log(error) }) // 请求成功

// 失败请求案例
fn(2).then((data)=>{
  console.log(data)
  },(error)=>{
    console.log(error)
  }) // 请求失败

还是以上面的案例为例，

Promise.all([fn(), fn()]).then((data)=>{
    console.log(data)
},(error)=>{
    console.log(error)
}) // 返回一个数组：["请求成功", "请求成功"]

Promise.race([fn(), fn()]).then((data)=>{
    console.log(data)
},(error)=>{
    console.log(error)
}) // 返回第一个请求成功："请求成功"

共同点：它们失败都会返回第一个请求失败的数据

# 3. 必考：手写函数防抖和函数节流

函数防抖

概念: 任务频繁触发的情况下，只有任务触发的间隔超过指定间隔以后，下一个任务才会执行。

举例：以送外卖为例，在5秒内没有接到新单就开始送外卖，如下

function fn(){ console.log('开始送外卖') }
let timerId = null
buttons.onclick = function() {
    /* 判断外面时间 */
    if(timerId) {
        window.clearTimeout(timerId)
        console.log('上一单还没超过5秒！')
    }
    timerId = setTimeout(()=>{
        fn()
        timerId = null
    }, 5000)
}

函数节流

概念: 指定时间间隔内只会执行一次任务。

举例：以技能释放CD为例，释放技能后在3秒内不能再次释放

function fn(){ console.log('释放技能') }
let cd = false
buttons.onclick = function(){
    if(cd){
        console.log('技能正在冷却')
    }else {
        /* 点击时执行技能，并将 cd 变为 true，执行技能后过 3 秒 cd 才会再次变为 false。 */
        fn()
        cd = true
        let timerId = setTimeout(()=>{
            cd = false
        }, 3000)
    }
}

# 4. 必考：手写AJAX

概念：AJAX就是用JS发请求和收响应。浏览器在 window 上加了一个 XMLHttpRequest 函数，用这个函数可以构造一个对象，用于发送请求接收响应。

简化版：

var request = new XMLHttpRequest()
request.open('GET', '/xxx')
request.onload = ()=>{ console.log('请求成功') }
request.send()

完整版：

let request = new XMLHttpRequest()
request.open('get', 'http://xx.com/xxx')
request.send()
request.onreadystatechange = ()=>{
    if(request.readyState === 4){
        if(request.status >= 200 && request.status < 300){
            let string = request.responseText
            let object = window.JSON.parse(string)
            console.log('请求成功!')
        }else if(request.status >= 400){
            console.log('请求失败!')
        }
    }
}

# 6. 必考：闭包/立即执行函数是什么？

概念

  • 函数使用了函数外的变量就叫做「闭包」。
  • 声明后立即调用的函数就叫做「立即执行函数」。

闭包案例

let a = 1
function fn(){
    console.loga(a)
}

立即执行函数案例

!fucntion(){}()

立即执行函数 + 闭包

假设我们在做一个游戏，在写其中关于「还剩几条命」的代码。但生命值不能让别人访问到，只能做生命加减操作，如下：

!function(){
    var lives = 50
    window.奖励一条命 = function(){
        lives += 1
        console.log(lives)
    }
    window.死一条命 = function(){
        lives -= 1
        console.log(lives)
    }
}()

window.奖励一条命() // 51
window.死一条命() // 50

上面的代码就实现了 闭包 + 立即执行函数，我们只能通过调用 奖励一条命 或 死一条命 方法来控制 lives 变量，且由于函数在声明后立即调用，所以又构成了立即执行函数。

# 7. 必考：什么是 JSONP，什么是 CORS，什么是跨域？

什么是跨域？

域名 A 下的 JS 想通过 AJAX 请求域名 B 下的数据，就叫做跨域。

什么是 CORS？

CORS 就是用来解决跨域请求的一种方式，在响应头携带一个 Access-Control-Allow-Origin 字段，表明指定的域名可以访问该网站。

什么是 JSONP？

JSONP 就是由于当前浏览器不支持 CORS 或不兼容 CORS 等其他原因，这是我们需要跨域请求，就必须使用另外一种方式来跨域。于是我们就跨域请求一个 JS 文件，这个 JS 文件会执行一个回调，回调里面就有我们需要的数据，回调的名称是可以随机生成的，可以给一个随机数，然后把这个名字用一个名叫 callback 参数传给后台，后台会把这个函数再次返回给我们并执行。

优点: 兼容IE，它可以跨域。

缺点: 由于它是利用 script 标签获取，所以它只能发送 GET 请求，它不支持POST请求。

# 9. 常考：如何实现深拷贝？

关键词：递归、判断类型、检查循环引用、深拷贝不可能拷贝原型。

代码：

function deepCopy(target){ 
let copyed_objs = []    // 此数组解决了循环引用和相同引用的问题，它存放已经递归到的目标对象 
    function _deepCopy(target){ 
        if((typeof target !== 'object')||!target){return target;}
        for(let i= 0 ;i<copyed_objs.length;i++){
            if(copyed_objs[i].target === target){
                return copyed_objs[i].copyTarget;
            }
        }
        let obj = {};
        if(Array.isArray(target)){
            obj = []   // 处理target是数组的情况 
        }
        copyed_objs.push({target:target,copyTarget:obj}) 
        Object.keys(target).forEach(key=>{ 
            if(obj[key]){ return;} 
            obj[key] = _deepCopy(target[key]);
        }); 
        return obj;
    } 
    return _deepCopy(target);
}

# 11. 常考：不用 class 如何实现继承？用 class 又如何实现？

不用 class 实现继承

function Animal () {
  this.a = 10
}
Animal.prototype.move = function () {
  console.log(222)
}
function Dog () {
  Animal.apply(this, arguments)
  this.b = 20
}

let f = function() {}
f.prototype = Animal.prototype
Dog.prototype = new f()

Dog.prototype.constructor = Dog

Dog.say = function(){ console.log('汪汪汪') }

用 class 实现继承

function Animal () {
    this.a = 10
}
class Dog extends Animal {
    constructor () {
        // 注意: 必须先调用 super() 才能使用 "this"。
        super()
        this.b = 20
    }
    say () {
        console.log('汪汪汪')
    }
}

其中

  • extends：表示继承该方法。
  • constructor：在一个类中只能有一个名为 constructor 的特殊方法。

# 12. 常考：如何实现数组去重？

代码

  1. 使用 hash
  2. 使用 [...new Set(array)]
  3. WeakMap

1、使用hash

原理：通过对象 key 的唯一性实现去重。

function unique(array) {
    let obj = {}
    let base = []
    for(let i = 0; i < array.length; i++){
        if(!obj[array[i]]){
            base.push(array[i])
            obj[array[i]] = true
        }
    }
    return base
}
let a = [1, 2, 3, 4, 1, 2, 3, 4]
console.log(unique(a))

2、使用 Set(array)

利用 new Set(arr) 中的值都是唯一值的原理去重（返回一个伪数组）

第一种方式是使用剩余参数将去重后的伪数组加到新的数组中，转为数组。

let arr = [1, 1, 2, 2, 3, 3, null, null, NaN, NaN] 
let data = [...new Set(arr)]

第二种方式是使用 Array.from 将去重后的伪数组转为数组

let arr = [1, 1, 2, 2, 3, 3, null, null, NaN, NaN]
let data = Array.from(new Set(arr))

# 15. 必考：事件委托

错误写法

ul.addEventListener('click', function(e){
    if(e.target.tagName.toLowerCase() === 'li'){
        console.log('点击了li')
    }
})

正确写法

function xxx(element, eventType, selector, fn) {
    element.addEventListener(eventType, e => {
        let el = e.target // 获取当前事件元素
        while (!el.matches(selector)) {
            if (element === el) {
                el = null
                break
            }
            el = el.parentNode
        }
        el && fn.call(el, e, el)
    })
    return element
}

# 16. 曾考：用 mouse 事件写一个可拖曳的 div

html

<div id="drag"></div>

css

let dragging = false
let position = null

drag.addEventListener('mousedown', function (e) {
    dragging = true
    position = [e.clientX, e.clientY]
})

document.addEventListener('mousemove', function (e) {
    if (dragging === false) { return }
    const x = e.clientX
    const y = e.clientY
    const deltaX = x - position[0]
    const deltaY = y - position[1]
    const left = parseInt(drag.style.left || 0)
    const top = parseInt(drag.style.top || 0)
    // console.log(btns.style.left, left, top)
    drag.style.left = left + deltaX + 'px'
    drag.style.top = top + deltaY + 'px'
    position = [x, y]
})

document.addEventListener('mouseup', function () {
    dragging = false
})

# 冒泡排序

function bubbleSort(array){
    let lengths = array.length 
    for(let i=0; i<lengths-1; i++){
        for(let j=0; j<lengths-1-i; j++){
            if(array[j]<array[j+1]){
                [array[j], array[j+1]] = [array[j+1], array[j]]
            }
        }
    }
    return array
}
