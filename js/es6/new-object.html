# 创建对象

# 之前的创建方法

var a1 = new Object()
var a2 = {}

上面的两种方法创建的对象都有对象的原型链属性

# ES6新增方法

let b1 = Object.create(null) // 没有对象的原型链
let b2 = Object.create(Object.prototype) // 传入对象的原型链

ES6方法创建的对象，只有在传入 Object.prototype 是才会生成具有对象原型链的空对象 当传入的值为 null 时，生成的对象是不具有原型链的（也就是真正的空对象）

案例

var a = new Object()                     // ES5 - 有原型链的空对象
var b = Object.create(null)              // ES6 - 什么都没有的空对象(真正的空对象)
var c = Object.create(Object.prototype)  // ES6 - 有原型链的空对象
var d = {}                               // ES5 - 有原型链的空对象

# 使用变量作为对象属性名

let name = 'a'
let object = {
    [name]: 1
}
console.log(object) // 输出 {a: 1}

对象属性名可以做简单的运算

let n = 2
let a = {
    [n*n + n/n + n/3]: 1
}
console.log(a) // 输出 {5.666666666666667: 1}

# 设置对象读取/写入权限

在对象内部可以添加两个方法：get方法，负责读取时做的操作。set方法，负责修改时做的操作。

通过这两个方法可以对对象的属性做访问/修改的限制，如下

let age = 18
let o = {
    get age () {
        return age
    },
    set age (value) {
        if (value < 100) {
            age = value
        }else{
            age = 100
        }
    }
}
console.log(age)   // 输出 18
console.log(o.age) // 输出 18
o.age = 1000
console.log(o.age) // 输出 100

# 实现 a === 1 && a === 2 && a === 3 结果为 true

let i = 0
/* 给 window 添加一个属性，属性名是 a，a 有一个 get 函数 */
Object.defineProperty(window, 'a', {
    get(){
        i+=1
        return i
    },
})
a === 1 && a === 2 && a === 3 // 输出 true

原理

控制全局变量 a 的读取过程，每次访问 a 的值都会改变

# 对象浅拷贝

之前的写法

let obj1 = { a: 1, b: 2, c: 3 }
let obj2 = {}
for(let key in obj1){
    obj2[key] = obj1[key]
}
obj2 === obj1
console.log(obj1) // 输出 {a: 1, b: 2, c: 3}
console.log(obj2) // 输出 {a: 1, b: 2, c: 3}
obj2.a = 100 // obj2.a 为 100
obj1.a // obj1.a 为 1

ES6浅拷贝（1）

let obj1 = {a: 1, b: 2, c: 3}
let obj2 = Object.assign({}, obj1)
console.log(obj2) // {a: 1, b: 2, c: 3}

ES6浅拷贝（2）

let obj1 = { a: 1, b: 2, c: 3 }
let obj2 = {...obj1}
console.log(obj2)

ES6对象案例

let obj1 = { a: 1 }
let obj2 = { b: 2 }
let obj3 = { c: 3 }
let aa = 1
let bb = 2
let obj4 = {
    ...obj1, ...obj2, ...obj3, aa, bb, get x(){ return 'hi' }
}
obj4

在 obj4 中添加 obj1 obj2 obj3 对象的属性，并添加变量 aa bb 为 obj4 的属性，设置属性x的访问值为 'hi'

# 设置对象原型

使用 a = Object.create(b) 的方法可以给对象添加新的原型链。

let a = {}
a.__proto__
a.toString() // 输出 "[object Object]"
a.valueOf() // 输出 {}
a.__proto__ === Object.prototype // 输出 true
Object.prototype
let b = {
    sayHi(){
        console.log('hi')
    }
}
a = Object.create(b) // a的原型指向b
a.__proto__ === b // 输出 true

访问对象原型链

let a = {}
a.__proto__
Object.getPrototypeOf(a) // 推荐写法

查询对象的属性是否存在

a.hasOwnProperty('indexOf')

使用 hasOwnProperty('__proto__') 无法查看原型属性

a.hasOwnProperty('__proto__') // 输出 false
